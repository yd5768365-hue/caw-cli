    def _generate_optimization_report(self) -> str:
        """ç”Ÿæˆä¼˜åŒ–æŠ¥å‘Š"""
        if not self.results:

        report_lines = [

                f"è¿­ä»£ {result.iteration}: {params_str} | "

                f"è€—æ—¶: {result.analysis_time:.2f}s{marker}"

        report_lines.extend(
            [

                "æœ€ä½³ç»“æœ:",

            [
